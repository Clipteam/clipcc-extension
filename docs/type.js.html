/**
 * Block types.
 * @readonly
 * @enum {number}
 */
const BlockType = {
    /** A general command block */
    COMMAND: 1,
    /** A block which returns a number or string */
    REPORTER: 2,
    /** A block which returns a boolean value */
    BOOLEAN: 3,
    /** [Unsupported] A block with branch(es) */
    BRANCH: 4,
    /** [Unsupported] A hat block */
    HAT: 5
};

/**
 * Argument types.
 * @readonly
 * @enum {number}
 */
const ArgumentType = {
    /** Number */
    NUMBER: 1,
    /** String */
    STRING: 2,

    /** [Unsupported] */
    NOTE: 7,

};
